#include <iostream>
using namespace std;
int main()
{
    char ch;
    cout<<"enter the value for ch:\n";
    cin>>ch;
    ch=tolower(ch);
    if(ch=='a' || ch== 'e' || ch=='i' || ch=='o' || ch=='u')
    {
        cout<<"vowel";
    }
    else
    {
        cout<<"not a vowel";
    }
    return 0;
}
